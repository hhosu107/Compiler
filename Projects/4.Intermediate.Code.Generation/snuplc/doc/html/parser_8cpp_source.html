183  // so we don't have to save each return value of subroutineDecl.
184  while(subroutineDecl(m) != NULL) ;
185

303

310  }
311

313  params->push_back(new CSymParam(size, name, ptrtype));

317  }

324  //
325  // subroutineDecl = (procedureDecl | functionDecl) subroutineBody ident ";"
326  // procedureDecl = "procedure" ident [formalParam] ";"
327  // functionDecl = "function" ident [formalParam] ":" type ";"
328  //
329  // FIRST(subroutineDecl) = {tProcedure, tFunction}
330  //

333 
334  CToken dummy;
335  bool isProcedure = false;
336 
337  // subroutineDecl -> "procedure" ...
338  if(tt == tProcedure){
339  Consume(tProcedure, &dummy);
340  isProcedure = true;
341  }
342 
343  // subroutineDecl -> "function" ...
344  else if(tt == tFunction){
345  Consume(tFunction, &dummy);
346  }
347

350  }

355  const CType* retType;
356  CTypeManager *tm = CTypeManager::Get();

360  SetError(idBegin, "this proc/func is redeclared");
361  }
362 
363  // subroutineDecl -> ... [ formalParam ] ...
364  // formalParam ::= "(" [ varDecl { ";" varDecl } ] ")"
365  vector<CSymParam*> params;
366  tt = _scanner->Peek().GetType();
367  if(tt == tLParen) { // formalParam
368  // formalParam -> "(" ...
369  Consume(tLParen);
370 
371  // formalParam -> ... [ varDecl ...
372  tt = _scanner->Peek().GetType();
373 
374  // Does a varDecl exist or not?
375  // Here, since it is for the formalParam,
376  // we have to save each formal parameters sequentially.
377  if(tt == tIdent){
378  // formalParam -> ... [ varDecl ...
379  symbols = varDecl(symbols, stParam, &params, true);
380 
381  // formalParam -> ... [ ... { ";" varDecl } ] ...
382  for(;;){
383  tt = _scanner->Peek().GetType();
384  if(tt != tSemicolon) break;
385  Consume(tSemicolon);
386  symbols = varDecl(symbols, stParam, &params, true);
387  }
388  }
389 
390  // formalParam -> ... ")"
391  Consume(tRParen);
392  }
393 
394  if(isProcedure){
395  // subroutineDecl -> ... ";" ...
396  retType = tm->GetNull();
397  Consume(tSemicolon);
398  }
399  else{
400  // subroutineDecl -> ... ":" type ";" ...
401  Consume(tColon);
402  retType = ReadType(false, false)->GetType();
403  Consume(tSemicolon);
404  }
405 
406  // make CSymProc* for this function and add it into parent symtab
407  CSymProc *symproc;
408  symproc = new CSymProc(idBegin.GetValue(), retType);
409  int idx = 0;
410  for(CSymParam* param : params){
411  symproc->AddParam(param);
412  }
413  (s->GetSymbolTable())->AddSymbol(symproc);
414 
415  m = new CAstProcedure(dummy, idBegin.GetValue(), s, symproc);
416  // 3rd element sets parent scope. In here, it is s.
417  // Still, its symbol table has to be set, in the below code.
418 
419  // subroutineDecl -> ... subroutineBody ...
420  // subroutineBody ::= varDeclaration "begin" statSequence "end"
421 
422  // subroutineBody -> varDeclaration ...
423  symbols = varDeclaration(symbols, stLocal, false);
424  vector<CSymbol*> symbolList = symbols->GetSymbols();
425  for(int i=0; i<symbolList.size(); i++){
426  (m->GetSymbolTable())->AddSymbol(symbolList[i]);
427  }
428 
429  // subroutineBody -> ... "begin" ...
430  Consume(tBegin);
431 
432  // subroutineDecl -> ... statSequence "end"
433  CAstStatement *statseq = NULL;
434  statseq = statSequence(m);
435  Consume(tEnd);
436 
437  // subroutineDecl -> ... ident ";"
438  Consume(tIdent, &idEnd);
439 
440  if(idBegin.GetValue() != idEnd.GetValue()){
441  string err = "procedure/function identifier mismatch ('";
442  err += idBegin.GetValue();
443  err += "' != '";
444  err += idEnd.GetValue();
445  err += "').";
446  SetError(idEnd, err);
447  }
448 
449  Consume(tSemicolon);
450 
451  m->SetStatementSequence(statseq);
452 
453  // Add this symbol into function itself to able recursive call?
454  // In the reference parser, it is not given to its symbol table.
455  // (m->GetSymbolTable())->AddSymbol(symproc);
456 
457  return m;
458 }

461 const CAstType* CParser::ReadType(bool openArray, bool beArray){
462  //
463  // type ::= basetype | type "[" [ number ] "]".
464  // <=>
465  // type ::= basetype type'
466  // type' ::= epsilon | "[" [ number ] "]" type'
467  //
468  CTypeManager *tm = CTypeManager::Get();
469  EToken tt = _scanner->Peek().GetType();
470  CToken typeToken;
471  const CType* datatype;
472 
473  // type -> basetype
474  if(tt == tBoolean){
475  Consume(tBoolean, &typeToken);
476  datatype = tm->GetBool();
477  }
478  else if(tt == tChar){
479  Consume(tChar, &typeToken);
480  datatype = tm->GetChar();
481  }
482  else if(tt == tInteger){
483  Consume(tInteger, &typeToken);
484  datatype = tm->GetInt();
485  }
486  else{
487  SetError(_scanner->Peek(), "basetype expected.");
488  return NULL;
489  }
490 
491  // type -> ... type'
492  // we used stack since we have to make CType* node from the last dimension
493  stack<long long> NElems;
494 
495  for(;;){ // array check
496  tt = _scanner->Peek().GetType();
497 
498  if(tt == tLBracket){
499  Consume(tLBracket);
500 
501  tt = _scanner->Peek().GetType();
502  if(!openArray || tt == tNumber){
503  CToken intToken;
504  Consume(tNumber, &intToken);
505  NElems.push(strtoll(intToken.GetValue().c_str(), NULL, 10));
506  }
507  else if(tt == tRBracket){
508  NElems.push(-1LL);
509  }
510  else{
511  SetError(_scanner->Peek(), "[number] \"]\" expected");
512  return NULL;
513  }
514 
515  Consume(tRBracket);
516  }
517  else break;
518  }
519 
520  if(!beArray && !(NElems.empty())){
521  SetError(typeToken, "invalid composite type for function.");
522  }
523 
524  // make CType* node iteratively
525  while(!(NElems.empty())){
526  long long size = NElems.top();
527  NElems.pop();
528  if(size >= 0){
529  datatype = tm->GetArray(size, datatype);
530  }
531  else{
532  datatype = tm->GetArray(CArrayType::OPEN, datatype);
533  }
534  }
535 
536  return new CAstType(typeToken, datatype);
537 }

542  //
543  // statSequence ::= [ statement { ";" statement } ].
544  // statement ::= assignment | subroutineCall | ifStatement |
545  // whileStatement | returnStatement.
